dnf clean all
dnf check-update

# Enable RPMfusion
dnf install https://download1.rpmfusion.org/free/fedora/rpmfusion-free-release-$(rpm -E %fedora).noarch.rpm https://download1.rpmfusion.org/nonfree/fedora/rpmfusion-nonfree-release-$(rpm -E %fedora).noarch.rpm


# Install Local Packages
dnf install *.rpm


# Remove Unnecessary software
dnf remove claws-mail* pragha* transmission* asunder* xscreensaver* abiword* gnumeric* goffice*


# Install Prime software
dnf groupinstall 'Development Tools'


# Install Better Fonts
dnf copr enable dawid/better_fonts
dnf install fontconfig-enhanced-defaults fontconfig-font-replacements


# Install Office
dnf install libreoffice


# Install Essential extras
dnf install keepassxc freshplayerplugin icedtea-web light-locker redshift-gtk hplip-gui simple-scan


# Install Multimedia
dnf install inkscape gimp vlc smplayer mpv smplayer-themes darktable rawtherapee kazam obs-studio audacity HandBrake-gui blender

dnf copr enable dirkdavidis/papirus-icon-theme
dnf install bluecurve-cursor-theme papirus-icon-theme pop-icon-theme numix-icon-theme-circle numix-gtk-theme arc-theme paper-icon-theme

git clone https://github.com/Jazqa/adwaita-tweaks.git ~/.themes/'Adwaita Tweaks'

dnf install spectre-meltdown-checker persepolis qbittorrent stellarium wireshark


# Install Games
dnf install steam aisleriot gnome-chess gnome-mines gnome-mahjongg gnome-sudoku
